def create_increasing_population(init_population, list_size, factor):
    """
    This function returns a numpy array with a increasing value 
    by factor "factor" of size size_list. The last value will be 
    the init_population value. 
    """
    population_list = [init_population]

    for i in range(list_size - 1):
        population_list.append(population_list[i] - (factor * population_list[i]))
        
    return np.flipud(np.array(population_list)) 

def extract_data(country, population_increase = 0):
    """
    This function returns for a given country name a list of GPD values, 
    happiness scores and a list of the years to which the two previous 
    lists correspond. A yearly increase in population can be simulated by setting
    the variable population_increase to a percentage (0.1 = 10% per year).
    """
    
    df_country = df_happiness.loc[df_happiness['Entity'] == country]
    GDP_values = df_GDP.loc[country].values.tolist()[-len(df_country.Year.values):]
    
    # Create a population array corresponding with/without a decrease in population
    population_array = create_increasing_population(df_population.loc[country].Population, 
                                                    len(GDP_values), population_increase)
    
    # Deviding the GDP score by populion to get GDP per capita
    GDP_values_capita = list(np.array(GDP_values) / population_array * 1000000)
    
    happiness_values = df_country.Score.values.tolist()
    years = years_of_interest[-len(df_country.Year.values):]
    
    return GDP_values_capita, happiness_values, years
    
def plot_correlation(GDP_values, happiness_values, country, years):
    """
    This function plots the correlation between GPD values and happiness
    scores for a given country. It also places year labels at every datapoint.
    """
    
    plt.plot(GDP_values, happiness_values)
    plt.title("Happiness score per GDP per capita of " + country)
    plt.xlabel("GDP per capita in International Dollars")
    plt.ylabel("Happiness score")

    # Add year labels to plot
    for i, (x,y) in enumerate(zip(GDP_values,happiness_values)):
        label = years[i]
        plt.annotate(label, (x,y), textcoords="offset points", xytext=(0,10), ha='center')
    # plt.savefig(country + ".png")
    plt.show()
    
def find_correlation_values(country_list, increase):
    """
    This function returns a list of all correlation values an a list of all
    corresponding p_values for a list of country names.
    """
    
    correlation_values = []
    p_values = []

    for country in country_list:
        GDP_values, happiness_values, years = extract_data(country)

       
        if len(GDP_values) > 1 and not increase:
            correlation, p_value = pearsonr(GDP_values, happiness_values)
            correlation_values.append(correlation)
            p_values.append(p_value)
                
        elif len(GDP_values) > 1 and np.all(np.diff(GDP_values) >= 0) and increase:
            correlation, p_value = pearsonr(GDP_values, happiness_values)
            correlation_values.append(correlation)
            p_values.append(p_value)
            

    return correlation_values, p_values

def plot_histogram(correlation_values, bin_size = 0.1):
    """
    This functions plots a histogram given a list of correlation values and 
    a given bin_size (default is 0.1).
    """
    
    plt.xlim([min(correlation_values), max(correlation_values)])
    plt.hist(correlation_values, bins=np.arange(-1, 1, bin_size), alpha=0.5)
    plt.title('Histogram of correlation values')
    plt.xlabel('Correlation value')
    plt.ylabel('Count')
    # plt.savefig("Histogram.png")
    plt.show()
